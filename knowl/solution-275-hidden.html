\begin{equation*} \parbox{0.4\textwidth}{ \ \begin{aligned} \int_{1}^{7} \frac 1 {x^2} dx \amp = \int_{1}^{7} x^{-2} dx \\\\ \amp = \left (-1)x^{-1} \right|_{1}^{7} = \left \frac{-1}{x} \right|_{1}^{7} \\\\ \amp = \frac{-1}{7} - \frac{-1}{1} = \frac 1 {1} - \frac 1 {7} = 0.857143 \end{aligned} } \end{equation*}
